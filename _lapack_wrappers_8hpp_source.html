53  T *d, int64_t mode, T cond, T dmax, int64_t kl, int64_t ku,
54  char pack, C *A, int64_t lda) {
55  if (sizeof(int64_t) > sizeof(lapack_int)) {
56  assert(std::abs(m) <= std::numeric_limits<lapack_int>::max());
57  assert(std::abs(n) <= std::numeric_limits<lapack_int>::max());
58  assert(std::abs(kl) <= std::numeric_limits<lapack_int>::max());
59  assert(std::abs(ku) <= std::numeric_limits<lapack_int>::max());
60  assert(std::abs(lda) <= std::numeric_limits<lapack_int>::max());
61  assert(std::abs(mode) <= std::numeric_limits<lapack_int>::max());
62  }
63 
64  lapack_int m_ = (lapack_int) m;
65  lapack_int n_ = (lapack_int) n;
66  lapack_int kl_ = (lapack_int) kl;
67  lapack_int ku_ = (lapack_int) ku;
68  lapack_int lda_ = (lapack_int) lda;
69  lapack_int mode_ = (lapack_int) mode;
70 
71  std::vector<C> work(3 * std::max(m, n));
72 
73 #ifndef HCORE_WITH_LAPACK_ILP64
74  // 32-bit copy
75  std::vector<lapack_int> iseed_(&iseed[0], &iseed[(4)]);
76  lapack_int *iseed_ptr = &iseed_[0];
77 #else
78  lapack_int* iseed_ptr = iseed;
79 #endif
80 
81  lapack_int info_ = 0;
82 
83  if constexpr(is_complex<C>()) {
84  if constexpr(is_double<T>()) {
85  LAPACK_zlatms(
86  &m_, &n_, &dist, iseed_ptr, &sym, d, &mode_, &cond, &dmax, &kl_, &ku_,
87  &pack, (lapack_complex_double *) A, &lda_,
88  (lapack_complex_double *) &work[0], &info_ HCOREPP_LATMS_ENDING);
89  } else {
90  LAPACK_clatms(
91  &m_, &n_, &dist, iseed_ptr, &sym, d, &mode_, &cond, &dmax, &kl_, &ku_,
92  &pack, (lapack_complex_float *) A, &lda_,
93  (lapack_complex_float *) &work[0], &info_ HCOREPP_LATMS_ENDING);
94  }
95  } else {
96  if constexpr (is_double<T>()) {
97  LAPACK_dlatms(
98  &m_, &n_, &dist, iseed_ptr, &sym, d, &mode_, &cond, &dmax, &kl_, &ku_,
99  &pack, A, &lda_, &work[0], &info_ HCOREPP_LATMS_ENDING);
100  } else {
101  LAPACK_slatms(
102  &m_, &n_, &dist, iseed_ptr, &sym, d, &mode_, &cond, &dmax, &kl_, &ku_,
103  &pack, A, &lda_, &work[0], &info_ HCOREPP_LATMS_ENDING);
104  }
105  }
106  if (info_ != 0) {
107  throw std::runtime_error("lapack_latms error " + std::to_string(info_));
108  }
109 
110 #ifndef HCORE_WITH_LAPACK_ILP64
111  std::copy(iseed_.begin(), iseed_.end(), iseed);
112 #endif
113 }
